\section*{Workstream 1 --- One explicit global parameter hierarchy}

\textbf{Objective}: give a single parameter schedule with a strict order of choices and a final inequality chain where every error term is visibly $o(1)$.

\tealc{Implemented in TeX:} see \verb|\label{sec:parameter-schedule}| directly after \texttt{thm:spine-quantitative}.

\subsection*{Required parameter list}
\begin{itemize}
\item \textbf{Geometric mesh}: $h_j\downarrow 0$ (cell diameter).
\item \textbf{Transverse quantization}: $\delta_j$ with $\delta_j=o(h_j)$.
\item \textbf{Angle/graph control}: $\varepsilon_j\downarrow 0$.
\item \textbf{Holomorphic scale}: $M_j\to\infty$ with $h_j\lesssim M_j^{-1/2}$.
\item \textbf{Homology multiple}: integer $m$ (fixed once).
\end{itemize}

\subsection*{What must be proved under that schedule}
\begin{itemize}
\item[(i)] \textbf{Fixed class}: $[T_j]=\PD(m[\gamma])$ for all $j$.
\item[(ii)] \textbf{Gluing mass vanishes}: $\Mass(G_j)\to 0$.
\item[(iii)] \textbf{Budget accuracy}: $\Mass(S_j)\le c_0+o(1)$.
\end{itemize}

\violet{Parameter schedule v1 (explicit, with two regimes).}

\textbf{Global choice (once):} choose an integer $m\ge 1$ such that:
\begin{itemize}
\item $m[\gamma]\in H^{2p}(X,\Z)$ (clears denominators),
\item all periods $m\int_X\beta\wedge\eta_\ell\in\Z$,
\item $m$ is large enough for discrepancy rounding margin.
\end{itemize}

\textbf{Scale choice (sequence):} for $j=1,2,\dots$,
\begin{itemize}
\item $h_j := 2^{-j}$,
\item $M_j := \lceil C_0\,h_j^{-2}\rceil$,
\item $\varepsilon_j := \varepsilon_0$ constant in non-borderline regime $p<n/2$,
\item $\delta_j := h_j^{1+\alpha}$ with fixed $\alpha\in(0,1)$.
\end{itemize}

\textbf{Regime A ($p<n/2$):} With $\varepsilon_j=\varepsilon_0$ and H2 coherence estimate $\Delta_F\lesssim h_j^2$, the global weighted bound gives $\cF(\partial T^{\mathrm{raw}}_j)\to 0$.

\textbf{Regime B ($p=n/2$):} The naive exponent disappears. Commit to proving an explicit borderline closure lemma that supplies an extra small factor.

\section*{Workstream 2 --- H1: local holomorphic manufacturing}

\textbf{Objective}: make the local manufacturing package a clean theorem family with fully stated hypotheses and uniform constants.

\tealc{Implemented in TeX:} Added \texttt{prop:h1-package} at the spine-theorem point of use.

\violet{Finished H1 statement template:}

\textbf{Theorem (H1: uniform holomorphic multi-patch manufacturing on a cell).}
Let $(X,\omega)$ be projective K\"ahler, fix $p$, and fix $\varepsilon\in(0,\varepsilon_*)$.
There exist constants $c,C>0$ and $M_0(\varepsilon)$ such that for every integer $M\ge M_0(\varepsilon)$, every holomorphic chart identifying a neighborhood of a cell $Q$ with a subset of $\C^n$ and satisfying $\mathrm{diam}(Q)\le c\,M^{-1/2}$, every calibrated complex $(n-p)$-plane $P$, and every finite set of translations $\{t_a\}_{a=1}^N\subset P^\perp$ with separation $\|t_a-t_b\|\ge 10\,\varepsilon\,\mathrm{diam}(Q)$, there exist holomorphic complete intersections $Y^1,\dots,Y^N\in |L^M|$ such that:
\begin{itemize}
\item each $Y^a\cap Q$ is a single $C^1$ graph over $P+t_a$ with slope $\le C\varepsilon$,
\item the pieces are pairwise disjoint on $Q$,
\item and $\Mass([Y^a]\llcorner Q)=(1+O(\varepsilon^2))\,\Mass([P+t_a]\llcorner Q)$ uniformly.
\end{itemize}

\section*{Workstream 3 --- H2: global face coherence}

\textbf{Objective}: extract the global face coherence step into one crisp, checkable proposition.

\tealc{Implemented in TeX:} Added \texttt{prop:h2-package} at the spine-theorem point of use.

\violet{Finished H2 statement template:}

\textbf{Proposition (H2: global face coherence on a mesh).}
Fix $h>0$ and a smoothly rounded cubulation $\{Q\}$ of $X$ of mesh $h$.
Then there exists an integer choice of activation data producing a raw calibrated sheet-sum
\[
S=\sum_Q S_Q
\]
such that:
\begin{itemize}
\item \textbf{(H2.1) per-face template control}: on every interior face $F=Q\cap Q'$, the induced face slices admit a matching with displacement $\Delta_F\lesssim h^2$,
\item \textbf{(H2.2) piece boundary budget}: pieces satisfy the uniform slice boundary inequality,
\item \textbf{(H2.3) cohomology rounding compatibility}: period constraints satisfied up to $<1/2$.
\end{itemize}
Consequently, $\cF(\partial S)\to 0$ as $h\to 0$.

\section*{Workstream 5 --- The borderline $p=n/2$ case}

\textbf{Objective}: remove the ``loss of slack'' vulnerability at $p=n/2$.

\textbf{Why mandatory}: In the weighted scaling exponent $2-\frac{2n}{k}=\frac{n-2p}{n-p}$, the $h$-decay disappears when $k=n$.

\tealc{Implemented in TeX:} \texttt{lem:borderline-p-half} (routes $p=n/2$ through \texttt{prop:integer-transport}).

\violet{Explicit borderline closure lemma:}

\textbf{Lemma (Borderline closure at $p=n/2$).}
Let $p=n/2$ so $k=n$. Assume the strengthened face coherence estimate:
\[
\Delta_F\ \lesssim\ h^{2+\alpha_*}
\qquad\text{for some fixed }\alpha_*>0.
\]
Then the weighted global estimate yields $\cF(\partial T^{\mathrm{raw}})\to 0$ as $h\to 0$, hence the defect $\Def_{\mathrm{cal}}(T_j)\to 0$.

\textbf{Candidate mechanisms:}
\begin{itemize}
\item Second-order symmetry/cancellation
\item Closedness-driven cancellation (exploit $d\beta=0$)
\item Mismatch support thinning
\end{itemize}
