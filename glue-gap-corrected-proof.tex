\noindent
This note provides a self-contained corrected proof of the ``Microstructure/gluing estimate''
as it appears in \texttt{Hodge-v6-w-Jon-Update-MERGED.tex} (Proposition~\texttt{\textbackslash ref\{prop:glue-gap\}}).
The issue in the manuscript version is purely in the \emph{written proof} of the proposition:
the statement already contains the correct bound (displacement $\times$ slice mass), but the short proof sketch
incorrectly drops the slice-mass factor and references boundary faces rather than interior interfaces.

\medskip

\section*{Minimal background}
We use only the standard ``flat norm'' inequality: if $T$ is a current and $Q$ is any current with $\partial Q=T$,
then
\[
\F(T)\le \Mass(Q),
\]
since one may take $R=0$ in the definition $\F(T)=\inf\{\Mass(R)+\Mass(Q):T=R+\partial Q\}$.

\section*{Translation filling lemma}
\begin{lemma}[Flat-norm stability under translation]\label{lem:flat-translate}
Let $S$ be an integral $\ell$-cycle in $\mathbb R^d$ (so $\partial S=0$) with finite mass.
For any translation vector $v\in\mathbb R^d$, let $\tau_v(x)=x+v$ and $(\tau_v)_\#S$ be the pushforward.
Then there exists an integral $(\ell+1)$-current $Q$ such that
\[
\partial Q=(\tau_v)_\#S-S,
\qquad
\Mass(Q)\le \|v\|\,\Mass(S).
\]
\end{lemma}

\begin{proof}
Let $H:[0,1]\times\mathbb R^d\to\mathbb R^d$ be the straight-line homotopy $H(t,x)=x+t v$.
Define $Q:=H_\#([0,1]\times S)$.
Since $\partial([0,1]\times S)=\{1\}\times S-\{0\}\times S-[0,1]\times \partial S$ and $\partial S=0$,
we have $\partial Q=(\tau_v)_\#S-S$.
The map $H$ has Jacobian bounded by $\|v\|$ in the $t$-direction, hence $\Mass(Q)\le \|v\|\Mass(S)$.
\end{proof}

\section*{Corrected gluing estimate}
\begin{proposition}[Microstructure/gluing estimate (corrected proof)]\label{prop:glue-gap-corrected}
Let $k\ge 1$ and let
\[
T^{\mathrm{raw}}=\sum_Q S_Q
\]
be an integral $k$-current obtained by summing integral $k$-currents $S_Q$ supported on cells $Q$ of a cubulation.
Assume the boundary admits an oriented face decomposition
\[
\partial T^{\mathrm{raw}}=\sum_F B_F
\]
as a sum over interior interfaces $F=Q\cap Q'$ where each face mismatch is
\[
B_F = \bigl(\partial S_Q\bigr)\llcorner F - \bigl(\partial S_{Q'}\bigr)\llcorner F.
\]

Assume moreover that for each interface $F$ there exist cycle slices $\Sigma_F(u)$ on $F$ depending on
$u\in\Omega_F\subset\mathbb R^{2p}$ such that:
\begin{itemize}
\item for some multisets $\{u_{F,a}\}_{a=1}^{N_F}$ and $\{u'_{F,a}\}_{a=1}^{N_F}$,
  \[
  B_F=\sum_{a=1}^{N_F}\Sigma_F(u_{F,a})-\sum_{a=1}^{N_F}\Sigma_F(u'_{F,a});
  \]
\item in face coordinates, each $\Sigma_F(u)$ is obtained from $\Sigma_F(0)$ by translation by $u$;
\item there exists a matching $\sigma_F\in S_{N_F}$ with a uniform displacement bound
  \[
  \|u_{F,a}-u'_{F,\sigma_F(a)}\|\le \Delta_F\qquad \text{for all }a.
  \]
\end{itemize}

Define, for each $F$ and $a$, the translation vector $v_{F,a}:=u'_{F,\sigma_F(a)}-u_{F,a}$ and let $Q_{F,a}$
be the integral $k$-current provided by Lemma~\ref{lem:flat-translate} applied (in the face chart) to the
$(k-1)$-cycle $\Sigma_F(u_{F,a})$, so that
\[
\partial Q_{F,a}=\Sigma_F(u_{F,a})-\Sigma_F(u'_{F,\sigma_F(a)}),
\qquad
\Mass(Q_{F,a})\le \|v_{F,a}\|\,\Mass(\Sigma_F(u_{F,a})).
\]
Set
\[
Q_F:=\sum_{a=1}^{N_F} Q_{F,a},
\qquad
U:=\sum_F Q_F,
\qquad
T:=T^{\mathrm{raw}}-U.
\]
Then:
\begin{enumerate}
\item[\textnormal{(i)}] $U$ is an integral $k$-current and $\partial U=\partial T^{\mathrm{raw}}$, hence $T$ is an integral $k$-cycle.
\item[\textnormal{(ii)}] The mass of $U$ satisfies
  \[
  \Mass(U)\le \sum_F \Mass(Q_F)\le \sum_F \Delta_F\sum_{a=1}^{N_F}\Mass(\Sigma_F(u_{F,a})).
  \]
\item[\textnormal{(iii)}] In particular,
  \[
  \F(\partial T^{\mathrm{raw}})\le \Mass(U).
  \]
\end{enumerate}
\end{proposition}

\begin{proof}
Fix an interface $F$ and its matching $\sigma_F$.
Summing the boundary identities for $Q_{F,a}$ gives
\[
\partial Q_F=\sum_{a=1}^{N_F}\bigl(\Sigma_F(u_{F,a})-\Sigma_F(u'_{F,\sigma_F(a)})\bigr)=B_F.
\]
Summing over all interfaces yields
\[
\partial U=\sum_F \partial Q_F=\sum_F B_F=\partial T^{\mathrm{raw}},
\]
so $T=T^{\mathrm{raw}}-U$ is a cycle.

For the mass bound, by the triangle inequality and the estimate from Lemma~\ref{lem:flat-translate},
\[
\Mass(Q_F)\le \sum_{a=1}^{N_F}\Mass(Q_{F,a})
\le \sum_{a=1}^{N_F}\|u_{F,a}-u'_{F,\sigma_F(a)}\|\,\Mass(\Sigma_F(u_{F,a}))
\le \Delta_F\sum_{a=1}^{N_F}\Mass(\Sigma_F(u_{F,a})).
\]
Summing over $F$ gives (ii).
Finally, since $\partial U=\partial T^{\mathrm{raw}}$, taking $Q:=U$ and $R:=0$ in the flat norm definition yields
$\F(\partial T^{\mathrm{raw}})\le \Mass(U)$, proving (iii).
\end{proof}
